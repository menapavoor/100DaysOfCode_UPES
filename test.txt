# Q128: Count vowels and consonants in a file

file_name = "text.txt"

vowels = 0
consonants = 0

try:
    with open(file_name, "r") as f:
        text = f.read()
        for ch in text:
            if ch.isalpha():  # only process alphabets
                if ch.lower() in "aeiou":
                    vowels += 1
                else:
                    consonants += 1

    print("Vowels:", vowels)
    print("Consonants:", consonants)

except FileNotFoundError:
    print(f"Error: Could not open file {file_name}")
